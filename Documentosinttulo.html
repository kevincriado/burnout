                <p class="text-center text-[#4682B4] mb-8 max-w-4xl mx-auto">La comprensión del burnout ha evolucionado.                         <p class="text-sm">Proceso de desgaste en "grandes triunfadores" por estrés y altos ideales. El foco está en la <span class="font-bold">personalidad</span>.</p>

                        <p class="text-sm">Síndrome multidimensional como respuesta a estresores <span class="font-bold">interpersonales</span> crónicos en el trabajo.</p>

                    </div>

                    <div class="border-t-4 border-[#007BFF] p-4">

                        <p class="text-sm">Fenómeno <span class="font-bold">ocupacional</span> por estrés laboral crónico no gestionado. No es una enfermedad, es un riesgo laboral.</p>

        function simpleMarkdownToHtml(text) {

            let html = text

                .replace(/\*\*(.*?)\*\*/g, '<strong>$1</strong>')

                .replace(/\*(.*?)\*/g, '<em>$1</em>');

            html = html.split('\n').map(line => {

                if (line.trim().startsWith('- ') || line.trim().startsWith('* ')) {

                    return `<ul><li class="ml-4 list-disc">${line.trim().substring(2)}</li></ul>`;

                }

                return line ? `<p class="mb-2">${line}</p>` : '<br>';

            }).join('').replace(/<\/ul><ul>/g, '');

            return html;

        }
